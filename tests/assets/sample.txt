settings:
  fps = 30 + 3 *(5+5)
  render_mode = 'PathTracing'
  num_scenes = 100
  root_path = '/path/to/root'

stage:  
  open '*/scenario.usd'

  brake_models = fetch 'usd' from '*/models/' match 'brake_'
  
  obj = create 5 from brake_models :
    translate to <0,0,0>
    translate x to 50
    rotate <120,0,90> xyz
    scale <2, 1, 2>
    semantics 'class:obj'
    collider
    kinematics:
      velocity 150
  
  create Disk Light:
    radius:float 100
    
    every frame:
      camera_translate to <0,0, 250>


  camera = get camera at '/World/Room':
    exclude '/World/Room/CameraTop'

  edit camera:
    translate to <0,0,0>
    rotate 90
    rotate_around obj:
      radius Normal(100,5)

  every $num_scenes / 2 frame:
    edit obj:
      visibility Choice([true, false])


    instantiate 5 from '*/props':
      translate to Uniform(<-10,-10,10>, <10,10,10>)

writers:
  intrinsics = [[0, 0, 0], 
                [0, 0, 0], 
                [0, 0, 0]]
  
  DOPEWriter:
    pose = true
    intrinsics = intrinsics

  KittiWriter:
    bounding_box_2d = true
